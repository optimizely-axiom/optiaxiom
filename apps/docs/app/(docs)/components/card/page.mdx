### Anatomy

```tsx
import {
  Card,
  CardAction,
  CardCheckbox,
  CardContent,
  CardDescription,
  CardImage,
  CardLink,
  CardOverflow,
  CardTitle,
} from "@optiaxiom/react/unstable";

export default () => (
  <Card>
    <CardOverflow>
      <CardAction>
        <CardCheckbox />
      </CardAction>

      <CardImage />
    </CardOverflow>

    <CardImage />

    <CardContent>
      <CardTitle>
        <CardLink />
      </CardTitle>

      <CardDescription />

      <CardAction />
    </CardContent>
  </Card>
);
```
