# Select

Single select combobox widget to allow selection from a fixed set of options.

## Documentation

### Usage

<Demo component="select/usage" />

### Anatomy

```tsx
import {
  Select,
  SelectContent,
  SelectGroup,
  SelectItem,
  SelectLabel,
  SelectRadioItem,
  SelectSeparator,
  SelectTrigger,
} from "@optiaxiom/react";

export default () => (
  <Select>
    <SelectTrigger />

    <SelectContent>
      <SelectItem />
      <SelectRadioItem />

      <SelectGroup>
        <SelectLabel />
        <SelectItem />
        <SelectRadioItem />
      </SelectGroup>

      <SelectSeparator />
    </SelectContent>
  </Select>
);
```
