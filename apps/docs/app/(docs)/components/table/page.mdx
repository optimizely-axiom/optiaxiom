### Anatomy

```tsx
import {
  Table,
  TableBody,
  TableCell,
  TableHeader,
  TableHeaderCell,
  TableRow,
} from "@optiaxiom/react/unstable";

export default () => (
  <Table>
    <TableHeader>
      <TableRow>
        <TableHeaderCell />
      </TableRow>
    </TableHeader>

    <TableBody>
      <TableRow>
        <TableCell>
          <TableAction />
        </TableCell>
      </TableRow>
    </TableBody>
  </Table>
);
```
